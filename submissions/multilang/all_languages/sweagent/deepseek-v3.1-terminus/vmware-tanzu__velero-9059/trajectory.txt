   216	// ensureSingleDefaultBSL ensures that there is only one default BSL in the namespace.
   217	// the default BSL priority is as follows:
   218	// 1. follow the user's setting (the most recent validation BSL is the default BSL)
   219	// 2. follow the server's setting ("velero server --default-backup-storage-location")
   220	func (r *backupStorageLocationReconciler) ensureSingleDefaultBSL(locationList velerov1api.BackupStorageLocationList) (bool, error) {
   221	        // get all default BSLs
   222	        var defaultBSLs []*velerov1api.BackupStorageLocation
   223	        var defaultFound bool
   224	        for i, location := range locationList.Items {
   225	                if location.Spec.Default {
   226	                        defaultBSLs = append(defaultBSLs, &locationList.Items[i])
   227	                }
   228	        }
   229	
   230	        if len(defaultBSLs) > 1 { // more than 1 default BSL
   231	                // find the most recent updated default BSL
   232	                var mostRecentCreatedBSL *velerov1api.BackupStorageLocation
   233	                defaultFound = true
   234	                for _, bsl := range defaultBSLs {
   235	                        if mostRecentCreatedBSL == nil {
   236	                                mostRecentCreatedBSL = bsl
   237	                                continue
   238	                        }
   239	                        // For lack of a better way to compare timestamps, we use the CreationTimestamp
   240	                        // it cloud not really find the most recent updated BSL, but it is good enough for now
   241	                        bslTimestamp := bsl.CreationTimestamp
   242	                        mostRecentTimestamp := mostRecentCreatedBSL.CreationTimestamp
   243	                        if mostRecentTimestamp.Before(&bslTimestamp) {
   244	                                mostRecentCreatedBSL = bsl
   245	                        }
   246	                }
   247	
   248	                // unset all other default BSLs
   249	                for _, bsl := range defaultBSLs {
   250	                        if bsl.Name != mostRecentCreatedBSL.Name {
   251	                                bsl.Spec.Default = false
   252	                                if err := r.client.Update(r.ctx, bsl); err != nil {
   253	                                        return defaultFound, errors.Wrapf(err, "failed to unset default backup storage location %q", bsl.Name)
   254	                                }
   255	                                r.log.Debugf("update default backup storage location %q to false", bsl.Name)
   256	                        }
   257	                }
   258	        } else if len(defaultBSLs) == 0 { // no default BSL
   259	                defaultFound = false
   260	        } else { // only 1 default BSL
   261	                defaultFound = true
   262	        }
   263	        return defaultFound, nil
   264	}
