   205	// ListenQUICAndAssociate listens for quic connections with the provided `tlsConf.NextProtos` ALPNs on `addr`. The same addr can be shared between
   206	// different ALPNs.
   207	// The QUIC Transport used for listening is tagged with the `association`. Any subsequent `TransportWithAssociationForDial`,
   208	// or `DialQUIC` calls with the same `association` will reuse the QUIC Transport used by this method.
   209	// A common use of associations is to ensure /quic dials use the quic listening address and /webtransport dials use the
   210	// WebTransport listening address.
   211	func (c *ConnManager) ListenQUICAndAssociate(association any, addr ma.Multiaddr, tlsConf *tls.Config, allowWindowIncrease func(conn *quic.Conn, delta uint64) bool) (Listener, error) {
   212	        netw, host, err := manet.DialArgs(addr)
   213	        if err != nil {
   214	                return nil, err
   215	        }
   216	        laddr, err := net.ResolveUDPAddr(netw, host)
   217	        if err != nil {
   218	                return nil, err
   219	        }
   220	
   221	        c.quicListenersMu.Lock()
   222	        defer c.quicListenersMu.Unlock()
   223	
   224	        key := laddr.String()
   225	        entry, ok := c.quicListeners[key]
   226	        if !ok {
   227	                tr, err := c.transportForListen(association, netw, laddr)
   228	                if err != nil {
