   358	func GenerateISO9660(instDir, name string, instConfig *limayaml.LimaYAML, udpDNSLocalPort, tcpDNSLocalPort int, guestAgentBinary, nerdctlArchive string, vsockPort int, virtioPort string) error {
   359	        args, err := templateArgs(true, instDir, name, instConfig, udpDNSLocalPort, tcpDNSLocalPort, vsockPort, virtioPort)
   360	        if err != nil {
   361	                return err
   362	        }
   363	
   364	        if err := ValidateTemplateArgs(args); err != nil {
   365	                return err
   366	        }
   367	
   368	        layout, err := ExecuteTemplateCIDataISO(args)
   369	        if err != nil {
   370	                return err
   371	        }
   372	
   373	        for i, f := range instConfig.Provision {
   374	                switch f.Mode {
   375	                case limayaml.ProvisionModeSystem, limayaml.ProvisionModeUser, limayaml.ProvisionModeDependency:
   376	                        layout = append(layout, iso9660util.Entry{
   377	                                Path:   fmt.Sprintf("provision.%s/%08d", f.Mode, i),
   378	                                Reader: strings.NewReader(f.Script),
   379	                        })
   380	                case limayaml.ProvisionModeData:
   381	                        layout = append(layout, iso9660util.Entry{
   382	                                Path:   fmt.Sprintf("provision.%s/%08d", f.Mode, i),
   383	                                Reader: strings.NewReader(*f.Content),
   384	                        })
   385	                case limayaml.ProvisionModeBoot:
   386	                        continue
   387	                case limayaml.ProvisionModeAnsible:
   388	                        continue
   389	                default:
   390	                        return fmt.Errorf("unknown provision mode %q", f.Mode)
   391	                }
   392	        }
   393	
   394	        if guestAgentBinary != "" {
   395	                var guestAgent io.ReadCloser
   396	                if strings.HasSuffix(guestAgentBinary, ".gz") {
   397	                        logrus.Debugf("Decompressing %s", guestAgentBinary)
   398	                        guestAgentGz, err := os.Open(guestAgentBinary)
   399	                        if err != nil {
   400	                                return err
   401	                        }
   402	                        defer guestAgentGz.Close()
   403	                        guestAgent, err = gzip.NewReader(guestAgentGz)
   404	                        if err != nil {
   405	                                return err
   406	                        }
   407	                } else {
   408	                        guestAgent, err = os.Open(guestAgentBinary)
   409	                        if err != nil {
   410	                                return err
   411	                        }
   412	                }
   413	
   414	                defer guestAgent.Close()
   415	                layout = append(layout, iso9660util.Entry{
   416	                        Path:   "lima-guestagent",
   417	                        Reader: guestAgent,
   418	                })
   419	        }
   420	
   421	        if nerdctlArchive != "" {
   422	                nftgz := args.Containerd.Archive
   423	                nftgzR, err := os.Open(nerdctlArchive)
   424	                if err != nil {
   425	                        return err
   426	                }
   427	                defer nftgzR.Close()
   428	                layout = append(layout, iso9660util.Entry{
   429	                        // ISO9660 requires len(Path) <= 30
   430	                        Path:   nftgz,
   431	                        Reader: nftgzR,
   432	                })
   433	        }
   434	
   435	        if args.VMType == limayaml.WSL2 {
   436	                layout = append(layout, iso9660util.Entry{
   437	                        Path:   "ssh_authorized_keys",

   439	                })
   440	                return writeCIDataDir(filepath.Join(instDir, filenames.CIDataISODir), layout)
   441	        }
   442	
   443	        return iso9660util.Write(filepath.Join(instDir, filenames.CIDataISO), "cidata", layout)
   444	}

   479	func writeCIDataDir(rootPath string, layout []iso9660util.Entry) error {
   480	        slices.SortFunc(layout, func(a, b iso9660util.Entry) int {
   481	                return strings.Compare(strings.ToLower(a.Path), strings.ToLower(b.Path))
   482	        })
   483	
   484	        if err := os.RemoveAll(rootPath); err != nil {
   485	                return err
   486	        }
   487	
   488	        for _, e := range layout {
   489	                if dir := path.Dir(e.Path); dir != "" && dir != "/" {
   490	                        if err := os.MkdirAll(filepath.Join(rootPath, dir), 0o700); err != nil {
   491	                                return err
   492	                        }
   493	                }
   494	                f, err := os.OpenFile(filepath.Join(rootPath, e.Path), os.O_CREATE|os.O_RDWR, 0o700)
   495	                if err != nil {
   496	                        return err
   497	                }
   498	                if _, err := io.Copy(f, e.Reader); err != nil {
   499	                        _ = f.Close()
   500	                        return err
   501	                }
   502	                _ = f.Close()
   503	        }
   504	
   505	        return nil
