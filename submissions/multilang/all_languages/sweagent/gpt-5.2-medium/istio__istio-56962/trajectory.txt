   602	        // the "controller version". The original intent of this was to work around
   603	        // https://github.com/istio/istio/issues/44164, where we needed to transition from a global owner

   611	        // * 1.17 and older: version 1 OR no version at all, depending on patch release
   612	        // * 1.18+: version 5

   614	        // 2, 3, and 4 were intentionally skipped to allow for the (unlikely) event we need to insert

   619	// ManagedGatewayControllerVersion determines the version of the controller managing this Gateway,
   620	// and if we should manage this.
   621	// See ControllerVersionAnnotation for motivations.
   622	func ManagedGatewayControllerVersion(gw gateway.Gateway) (existing string, takeOver bool, manage bool) {
   623	        cur, f := gw.Annotations[ControllerVersionAnnotation]
   624	        if !f {
   625	                // No current owner, we should take it over.
   626	                return "", true, true
   627	        }
   628	        curNum, err := strconv.Atoi(cur)
   629	        if err != nil {
   630	                // We cannot parse it - must be some new schema we don't know about. We should assume we do not manage it.
   631	                // In theory, this should never happen, unless we decide a number was a bad idea in the future.
   632	                return cur, false, false
   633	        }
   634	        if curNum > ControllerVersion {
   635	                // A newer version owns this gateway, let them handle it
   636	                return cur, false, false
   637	        }
   638	        if curNum == ControllerVersion {
   639	                // We already manage this at this version
   640	                // We will manage it, but no need to attempt to apply the version annotation, which could race with newer versions
   641	                return cur, false, true
   642	        }
   643	        // We are either newer or the same version of the last owner - we can take over. We need to actually
   644	        // re-apply the annotation
   645	        return cur, true, true
   646	}
