    58	    int s, e; /* start-end indexes of { and } */
    59	
    60	    for (s = 0; s < keylen; s++)

    64	    if (s == keylen) return crc16(key, keylen) & 0x3FFF;

    66	    /* '{' found? Check if we have the corresponding '}'. */
