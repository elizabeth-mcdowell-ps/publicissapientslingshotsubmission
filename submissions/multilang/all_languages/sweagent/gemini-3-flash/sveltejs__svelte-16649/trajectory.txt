   166	                },

   169	                },

   176	async function common_setup(cwd: string, runes: boolean | undefined, config: RuntimeTest) {
   177	        const force_hmr = process.env.HMR && config.compileOptions?.dev !== false && !config.error;
   178	
   179	        const compileOptions: CompileOptions = {

   207	async function run_test_variant(
   208	        cwd: string,
   209	        config: RuntimeTest,
   210	        variant: 'dom' | 'hydrate' | 'ssr',
   211	        compileOptions: CompileOptions,
   212	        runes: boolean
   213	) {
   214	        let unintended_error = false;
   215	
   216	        let logs: string[] = [];
   217	        let warnings: string[] = [];
   218	        let errors: string[] = [];
   219	        let manual_hydrate = false;
   220	
   221	        {
   222	                // use some crude static analysis to determine if logs/warnings are intercepted.
   223	                // we do this instead of using getters on the `test` parameters so that we can
   224	                // squelch logs in SSR tests while printing temporary logs in other cases
   225	                let str = config.test?.toString() ?? '';
   226	                let n = 0;
   227	                let i = 0;
   228	                while (i < str.length) {
   229	                        if (str[i] === '(') n++;
   230	                        if (str[i] === ')' && --n === 0) break;
   231	                        i++;
   232	                }
   233	
   234	                if (str.slice(0, i).includes('logs')) {
   235	                        // eslint-disable-next-line no-console
   236	                        console.log = (...args) => {
   237	                                logs.push(...args);
   238	                        };
   239	                }
   240	
   241	                if (str.slice(0, i).includes('hydrate')) {
   242	                        manual_hydrate = true;
   243	                }
   244	
   245	                if (str.slice(0, i).includes('warnings') || config.warnings) {
   246	                        // eslint-disable-next-line no-console
   247	                        console.warn = (...args) => {
   248	                                if (typeof args[0] === 'string' && args[0].startsWith('%c[svelte]')) {
   249	                                        // TODO convert this to structured data, for more robust comparison?
   250	
   251	                                        let message = args[0];
   252	                                        message = message.slice(message.indexOf('%c', 2) + 2);
   253	
   254	                                        // Remove the "https://svelte.dev/e/..." link at the end
   255	                                        const lines = message.split('\n');
   256	                                        if (lines.at(-1)?.startsWith('https://svelte.dev/e/')) {
   257	                                                lines.pop();
   258	                                        }
   259	                                        message = lines.join('\n');
   260	
   261	                                        warnings.push(message);
   262	                                } else {
   263	                                        warnings.push(...args);
   264	                                }
   265	                        };
   266	                }
   267	
   268	                if (str.slice(0, i).includes('errors') || config.errors) {
   269	                        // eslint-disable-next-line no-console
   270	                        console.error = (...args) => {
   271	                                errors.push(...args);
   272	                        };
   273	                }
   274	        }
   275	
   276	        try {
   277	                unhandled_rejection = null;
   278	
   279	                if (config.expect_unhandled_rejections) {
   280	                        listeners.forEach((listener) => {
   281	                                // @ts-expect-error
   282	                                process.removeListener('unhandledRejection', listener);
   283	                        });
   284	                }
   285	
   286	                // hack to support transition tests
   287	                raf.reset();
   288	
   289	                // Put things we need on window for testing
   290	                const styles = globSync('**/*.css', { cwd: `${cwd}/_output/client` })
   291	                        .map((file) => fs.readFileSync(`${cwd}/_output/client/${file}`, 'utf-8'))
   292	                        .join('\n')
   293	                        .replace(/\/\*<\/?style>\*\//g, '');
   294	
   295	                window.location.href = '';
   296	                window.document.title = '';
   297	                window.document.head.innerHTML = styles ? `<style>${styles}</style>` : '';
   298	                window.document.body.innerHTML = '<main></main>';
   299	
   300	                window.addEventListener('error', (e) => {
   301	                        unhandled_rejection = e.error;
   302	                        e.preventDefault();
   303	                });
   304	
   305	                globalThis.requestAnimationFrame = globalThis.setTimeout;
   306	
   307	                let mod = await import(`${cwd}/_output/client/main.svelte.js`);
   308	
   309	                const target = window.document.querySelector('main') as HTMLElement;
   310	
   311	                let snapshot = undefined;
   312	
   313	                if (variant === 'hydrate' || variant === 'ssr') {
   314	                        config.before_test?.();
   315	                        // ssr into target
   316	                        const SsrSvelteComponent = (await import(`${cwd}/_output/server/main.svelte.js`)).default;
   317	                        const { html, head } = render(SsrSvelteComponent, {
   318	                                props: config.server_props ?? config.props ?? {},
   319	                                idPrefix: config.id_prefix
   320	                        });
   321	
   322	                        fs.writeFileSync(`${cwd}/_output/rendered.html`, html);
   323	                        target.innerHTML = html;
   324	
   325	                        if (head) {
   326	                                fs.writeFileSync(`${cwd}/_output/rendered_head.html`, head);
   327	                                window.document.head.innerHTML = window.document.head.innerHTML + head;
   328	                        }
   329	
   330	                        if (variant === 'hydrate') {
   331	                                // @ts-expect-error TODO
   332	                                if (config.snapshot) {
   333	                                        // @ts-expect-error
   334	                                        snapshot = config.snapshot(target);
   335	                                }
   336	                        }
   337	                } else {
   338	                        target.innerHTML = '';
   339	                }
   340	
   341	                if (variant === 'ssr') {
   342	                        if (config.ssrHtml) {
   343	                                assert_html_equal_with_options(target.innerHTML, config.ssrHtml, {
   344	                                        preserveComments:
   345	                                                config.withoutNormalizeHtml === 'only-strip-comments' ? false : undefined,
   346	                                        withoutNormalizeHtml: !!config.withoutNormalizeHtml
   347	                                });
   348	                        } else if (config.html) {
   349	                                assert_html_equal_with_options(target.innerHTML, config.html, {
   350	                                        preserveComments:
   351	                                                config.withoutNormalizeHtml === 'only-strip-comments' ? false : undefined,
   352	                                        withoutNormalizeHtml: !!config.withoutNormalizeHtml
   353	                                });
   354	                        }
   355	
   356	                        if (config.test_ssr) {
   357	                                await config.test_ssr({
   358	                                        logs,
   359	                                        // @ts-expect-error
   360	                                        assert: {
   361	                                                ...assert,
   362	                                                htmlEqual: assert_html_equal,
   363	                                                htmlEqualWithOptions: assert_html_equal_with_options
   364	                                        }
   365	                                });
   366	                        }
   367	                } else {
   368	                        logs.length = warnings.length = 0;
   369	
   370	                        config.before_test?.();

   373	                        let props: any;

   396	                                                target,
   397	                                                props,

   402	                        } else {
   403	                                instance = createClassComponent({
   404	                                        component: mod.default,

   409	                                        hydrate: variant === 'hydrate'
   410	                                });
   411	                        }
   412	
   413	                        if (config.error) {
   414	                                unintended_error = true;
   415	                                assert.fail('Expected a runtime error');
   416	                        }
   417	
   418	                        if (config.html) {
   419	                                flushSync();

   421	                                        preserveComments:
   422	                                                config.withoutNormalizeHtml === 'only-strip-comments' ? false : undefined,
   423	                                        withoutNormalizeHtml: !!config.withoutNormalizeHtml
   424	                                });
   425	                        }
   426	
   427	                        try {
   428	                                if (config.test) {
   429	                                        flushSync();

   432	                                                // wait for pending boundaries to render
   433	                                                await Promise.resolve();
   434	                                        }
   435	
   436	                                        await config.test({
   437	                                                // @ts-expect-error TS doesn't get it
   438	                                                assert: {
   439	                                                        ...assert,

   441	                                                        htmlEqualWithOptions: assert_html_equal_with_options

   445	                                                instance,

   447	                                                target,
   448	                                                snapshot,

   457	                                }
   458	
   459	                                if (config.runtime_error && !unhandled_rejection) {
   460	                                        unintended_error = true;
   461	                                        assert.fail('Expected a runtime error');
   462	                                }
   463	                        } finally {

   485	
   486	                                assert_html_equal(

   497	                                }
   498	                        }
   499	                }
   500	        } catch (err) {
   501	                if (config.runtime_error) {
   502	                        assert.include((err as Error).message, config.runtime_error);
   503	                } else if (config.error && !unintended_error) {
   504	                        assert.include((err as Error).message, config.error);
   505	                } else {
   506	                        throw err;
   507	                }
   508	        } finally {
   509	                config.after_test?.();

   511	                // Free up the microtask queue
   512	                // 1. Vitest's test runner which uses setInterval can log progress
   513	                // 2. Any expected unhandled rejections are ran before we reattach the listeners
   514	                await setImmediate();

   516	                if (config.expect_unhandled_rejections) {
   517	                        listeners.forEach((listener) => {
   518	                                // @ts-expect-error
   519	                                process.on('unhandledRejection', listener);
   520	                        });
   521	                }
   522	
   523	                console.log = console_log;
   524	                console.warn = console_warn;
   525	                console.error = console_error;
   526	
   527	                clear();
   528	        }
