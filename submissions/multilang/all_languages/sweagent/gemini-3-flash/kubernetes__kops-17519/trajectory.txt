+nvidiaGPU := context.Cluster.Spec.Containerd.NvidiaGPU
+if nvidiaGPU == nil || nvidiaGPU.Version == "" {

+continue

+if err := obj.SetField(podSpec, "spec", "template", "spec"); err != nil {
+return fmt.Errorf("error setting pod spec: %v", err)

+}
