   113	// Reconcile performs a full reconciliation for the object referred to by the Request.
   114	// The Controller will requeue the Request to be processed again if an error is non-nil or
   115	// Result.Requeue is true, otherwise upon completion it will remove the work from the queue.
   116	func (c *Controller) Reconcile(ctx context.Context, req reconcile.Request) (reconcile.Result, error) {
   117	        wl := &kueue.Workload{}
   118	
   119	        err := c.client.Get(ctx, req.NamespacedName, wl)
   120	        if err != nil {
   121	                return reconcile.Result{}, client.IgnoreNotFound(err)
   122	        }
   123	
   124	        log := ctrl.LoggerFrom(ctx)
   125	        log.V(2).Info("Reconcile Workload")
   126	
   127	        if !workload.HasQuotaReservation(wl) || workload.IsFinished(wl) || workload.IsEvicted(wl) {
   128	                return reconcile.Result{}, nil
   129	        }
   130	
   131	        provReqs := &autoscaling.ProvisioningRequestList{}
   132	        if err := c.client.List(ctx, provReqs, client.InNamespace(wl.Namespace), client.MatchingFields{RequestsOwnedByWorkloadKey: wl.Name}); client.IgnoreNotFound(err) != nil {
   133	                return reconcile.Result{}, err
   134	        }
   135	
   136	        // get the lists of relevant checks
   137	        relevantChecks, err := admissioncheck.FilterForController(ctx, c.client, wl.Status.AdmissionChecks, kueue.ProvisioningRequestControllerName)
   138	        if err != nil {
   139	                return reconcile.Result{}, err
   140	        }
   141	
   142	        checkConfig := make(map[kueue.AdmissionCheckReference]*kueue.ProvisioningRequestConfig, len(relevantChecks))
   143	        for _, checkName := range relevantChecks {
   144	                prc, err := c.helper.ConfigForAdmissionCheck(ctx, checkName)
   145	                if client.IgnoreNotFound(err) != nil {
   146	                        return reconcile.Result{}, err
   147	                }
   148	                checkConfig[checkName] = prc
   149	        }
   150	
   151	        activeOrLastPRForChecks := c.activeOrLastPRForChecks(ctx, wl, checkConfig, provReqs.Items)
   152	
   153	        wlInfo := workloadInfo{
   154	                checkStates: make([]kueue.AdmissionCheckState, 0),
   155	        }
   156	        err = c.syncCheckStates(ctx, wl, &wlInfo, checkConfig, activeOrLastPRForChecks)
   157	        if err != nil {
   158	                return reconcile.Result{}, err
   159	        }
   160	
   161	        err = c.deleteUnusedProvisioningRequests(ctx, provReqs.Items, activeOrLastPRForChecks)
   162	        if err != nil {
   163	                log.V(2).Error(err, "syncOwnedProvisionRequest failed to delete unused provisioning requests")
   164	                return reconcile.Result{}, err
   165	        }
   166	
   167	        err = c.syncOwnedProvisionRequest(ctx, wl, &wlInfo, checkConfig, activeOrLastPRForChecks)
   168	        if err != nil {
   169	                // this can also delete unneeded checks
   170	                log.V(2).Error(err, "syncOwnedProvisionRequest failed")
   171	                return reconcile.Result{}, err
   172	        }
   173	
   174	        return reconcile.Result{}, nil
   175	}
