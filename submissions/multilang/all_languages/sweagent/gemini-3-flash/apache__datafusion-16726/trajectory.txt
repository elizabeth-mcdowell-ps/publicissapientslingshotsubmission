   215	        // If the projection does not narrow the schema, we should not try to push it down:
   216	        let proj_exprs = projection.expr();
   217	        if proj_exprs.len() >= projection.input().schema().fields().len() {
   218	            return Ok(None);
   219	        }
   220	
   221	        let mut requirements = self.required_input_ordering().swap_remove(0);
   222	        if let Some(reqs) = requirements {
   223	            let mut updated_reqs = vec![];
   224	            let (lexes, soft) = reqs.into_alternatives();
   225	            for lex in lexes.into_iter() {
   226	                let Some(updated_lex) = update_ordering_requirement(lex, proj_exprs)?
   227	                else {
   228	                    return Ok(None);
   229	                };
   230	                updated_reqs.push(updated_lex);
   231	            }
   232	            requirements = OrderingRequirements::new_alternatives(updated_reqs, soft);
   233	        }
   234	
   235	        let dist_req = match &self.required_input_distribution()[0] {
   236	            Distribution::HashPartitioned(exprs) => {
   237	                let mut updated_exprs = vec![];
   238	                for expr in exprs {
   239	                    let Some(new_expr) = update_expr(expr, projection.expr(), false)?
   240	                    else {
   241	                        return Ok(None);
   242	                    };
   243	                    updated_exprs.push(new_expr);
   244	                }
   245	                Distribution::HashPartitioned(updated_exprs)
   246	            }
   247	            dist => dist.clone(),
   248	        };
   249	
   250	        make_with_child(projection, &self.input()).map(|input| {
   251	            let e = OutputRequirementExec::new(input, requirements, dist_req);
   252	            Some(Arc::new(e) as _)
   253	        })
   254	    }
   255	}

   257	impl PhysicalOptimizerRule for OutputRequirements {

   259	        &self,

   261	        _config: &ConfigOptions,

   263	        match self.mode {
   264	            RuleMode::Add => require_top_ordering(plan),

   269	                    {

   271	                    } else {
   272	                        Ok(Transformed::no(plan))
   273	                    }
   274	                })
   275	                .data(),
   276	        }
   277	    }

   280	        "OutputRequirements"
   281	    }

   284	        true
   285	    }
   286	}

   288	/// This functions adds ancillary `OutputRequirementExec` to the physical plan, so that
   289	/// global requirements are not lost during optimization.
   290	fn require_top_ordering(plan: Arc<dyn ExecutionPlan>) -> Result<Arc<dyn ExecutionPlan>> {
   291	    let (new_plan, is_changed) = require_top_ordering_helper(plan)?;
   292	    if is_changed {
   293	        Ok(new_plan)
   294	    } else {
   295	        // Add `OutputRequirementExec` to the top, with no specified ordering and distribution requirement.
   296	        Ok(Arc::new(OutputRequirementExec::new(
   297	            new_plan,
   298	            // there is no ordering requirement
   299	            None,
   300	            Distribution::UnspecifiedDistribution,
   301	        )) as _)
   302	    }
   303	}
   304	
   305	/// Helper function that adds an ancillary `OutputRequirementExec` to the given plan.
   306	/// First entry in the tuple is resulting plan, second entry indicates whether any
   307	/// `OutputRequirementExec` is added to the plan.
   308	fn require_top_ordering_helper(
   309	    plan: Arc<dyn ExecutionPlan>,
   310	) -> Result<(Arc<dyn ExecutionPlan>, bool)> {
   311	    let mut children = plan.children();
   312	    // Global ordering defines desired ordering in the final result.
   313	    if children.len() != 1 {
   314	        Ok((plan, false))
   315	    } else if let Some(sort_exec) = plan.as_any().downcast_ref::<SortExec>() {
   316	        // In case of constant columns, output ordering of the `SortExec` would
   317	        // be an empty set. Therefore; we check the sort expression field to
   318	        // assign the requirements.
   319	        let req_ordering = sort_exec.expr();
   320	        let req_dist = sort_exec.required_input_distribution().swap_remove(0);
   321	        let reqs = OrderingRequirements::from(req_ordering.clone());
   322	        Ok((
   323	            Arc::new(OutputRequirementExec::new(plan, Some(reqs), req_dist)) as _,
   324	            true,
   325	        ))
   326	    } else if let Some(spm) = plan.as_any().downcast_ref::<SortPreservingMergeExec>() {
   327	        let reqs = OrderingRequirements::from(spm.expr().clone());
   328	        Ok((
   329	            Arc::new(OutputRequirementExec::new(
   330	                plan,
   331	                Some(reqs),
   332	                Distribution::SinglePartition,
   333	            )) as _,
   334	            true,
   335	        ))
   336	    } else if plan.maintains_input_order()[0]
   337	        && (plan.required_input_ordering()[0]
   338	            .as_ref()
   339	            .is_none_or(|o| matches!(o, OrderingRequirements::Soft(_))))
   340	    {
   341	        // Keep searching for a `SortExec` as long as ordering is maintained,
   342	        // and on-the-way operators do not themselves require an ordering.
   343	        // When an operator requires an ordering, any `SortExec` below can not
   344	        // be responsible for (i.e. the originator of) the global ordering.
   345	        let (new_child, is_changed) =
   346	            require_top_ordering_helper(Arc::clone(children.swap_remove(0)))?;
   347	        Ok((plan.with_new_children(vec![new_child])?, is_changed))
   348	    } else {
   349	        // Stop searching, there is no global ordering desired for the query.
   350	        Ok((plan, false))
   351	    }
   352	}

   134	/// Calculates the required dimensions of the tree.
   135	/// This ensures we allocate enough space for the entire tree structure.
   136	///
   137	/// # Arguments
   138	/// * `plan` - The execution plan to measure
   139	///
   140	/// # Returns
   141	/// * A tuple of (width, height) representing the dimensions needed for the tree
   142	fn get_tree_width_height(plan: &dyn ExecutionPlan) -> (usize, usize) {
   143	    let children = plan.children();
   144	
   145	    // Leaf nodes take up 1x1 space
   146	    if children.is_empty() {
   147	        return (1, 1);
   148	    }
   149	
   150	    let mut width = 0;
   151	    let mut height = 0;
   152	
   153	    for child in children {
   154	        let (child_width, child_height) = get_tree_width_height(child.as_ref());
   155	        width += child_width;
   156	        height = cmp::max(height, child_height);
   157	    }
   158	
   159	    height += 1;
   160	
   161	    (width, height)
   162	}
