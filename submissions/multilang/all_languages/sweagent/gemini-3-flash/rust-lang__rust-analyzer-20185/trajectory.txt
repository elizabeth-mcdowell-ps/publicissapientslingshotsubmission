   241	    fn new(mut symbols: Box<[FileSymbol]>) -> SymbolIndex {
   242	        fn cmp(lhs: &FileSymbol, rhs: &FileSymbol) -> Ordering {
   243	            let lhs_chars = lhs.name.as_str().chars().map(|c| c.to_ascii_lowercase());
   244	            let rhs_chars = rhs.name.as_str().chars().map(|c| c.to_ascii_lowercase());
   245	            lhs_chars.cmp(rhs_chars)
   246	        }
   247	
   248	        symbols.par_sort_by(cmp);
   249	
   250	        let mut builder = fst::MapBuilder::memory();
   251	
   252	        let mut last_batch_start = 0;
   253	
   254	        for idx in 0..symbols.len() {
   255	            if let Some(next_symbol) = symbols.get(idx + 1) {
   256	                if cmp(&symbols[last_batch_start], next_symbol) == Ordering::Equal {
   257	                    continue;
   258	                }
   259	            }
   260	
   261	            let start = last_batch_start;
   262	            let end = idx + 1;
   263	            last_batch_start = end;
   264	
   265	            let key = symbols[start].name.as_str().to_ascii_lowercase();
   266	            let value = SymbolIndex::range_to_map_value(start, end);
   267	
   268	            builder.insert(key, value).unwrap();
   269	        }
   270	
   271	        let map = builder
   272	            .into_inner()
   273	            .and_then(|mut buf| {
   274	                fst::Map::new({
   275	                    buf.shrink_to_fit();
   276	                    buf
   277	                })
   278	            })
   279	            .unwrap();
   280	        SymbolIndex { symbols, map }
   281	    }

   347	        let ignore_underscore_prefixed = !self.query.starts_with("__");
   348	        while let Some((_, indexed_values)) = stream.next() {
   349	            for &IndexedValue { index, value } in indexed_values {
   350	                let symbol_index = &indices[index];
   351	                let (start, end) = SymbolIndex::map_value_to_range(value);
   352

   354	                    let non_type_for_type_only_query = self.only_types
   355	                        && !matches!(

   357	                            hir::ModuleDef::Adt(..)
   358	                                | hir::ModuleDef::TypeAlias(..)
   359	                                | hir::ModuleDef::BuiltinType(..)
   360	                                | hir::ModuleDef::TraitAlias(..)
   361	                                | hir::ModuleDef::Trait(..)
   362	                        );

   364	                        continue;
   365	                    }

   367	                    let symbol_name = symbol.name.as_str();

   370	                    }
