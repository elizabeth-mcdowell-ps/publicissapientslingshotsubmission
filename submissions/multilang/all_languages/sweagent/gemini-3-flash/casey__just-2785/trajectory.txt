   178	            function,
   179	            args: [a, b, c],
   180	            ..
   181	          } => {
   182	            let a = self.evaluate_expression(a)?;
   183	            let b = self.evaluate_expression(b)?;
   184	            let c = self.evaluate_expression(c)?;
   185	            function(function::Context::new(self, thunk.name()), &a, &b, &c)
   186	          }
   187	        };

   189	          function: thunk.name(),
   190	          message,
   191	        })
   192	      }
   193	      Expression::Concatenation { lhs, rhs } => {
   194	        let lhs = self.evaluate_expression(lhs)?;
   195	        let rhs = self.evaluate_expression(rhs)?;
   196	        Ok(lhs + &rhs)
   197	      }
   198	      Expression::Conditional {
   199	        condition,
   200	        then,
   201	        otherwise,
   202	      } => {
   203	        if self.evaluate_condition(condition)? {
   204	          self.evaluate_expression(then)
   205	        } else {
   206	          self.evaluate_expression(otherwise)
   207	        }
   208	      }
   209	      Expression::Group { contents } => self.evaluate_expression(contents),
   210	      Expression::Join { lhs: None, rhs } => Ok("/".to_string() + &self.evaluate_expression(rhs)?),
   211	      Expression::Join {
   212	        lhs: Some(lhs),
   213	        rhs,
   214	      } => {
   215	        let lhs = self.evaluate_expression(lhs)?;
   216	        let rhs = self.evaluate_expression(rhs)?;
   217	        Ok(lhs + "/" + &rhs)
   218	      }
   219	      Expression::Or { lhs, rhs } => {
   220	        let lhs = self.evaluate_expression(lhs)?;
   221	        if !lhs.is_empty() {
   222	          return Ok(lhs);
   223	        }
   224	        self.evaluate_expression(rhs)
   225	      }
   226	      Expression::StringLiteral { string_literal } => Ok(string_literal.cooked.clone()),
   227	      Expression::Variable { name, .. } => {
   228	        let variable = name.lexeme();

   230	          Ok(value.to_owned())
   231	        } else if let Some(assignment) = self
   232	          .assignments
   233	          .and_then(|assignments| assignments.get(variable))
   234	        {
   235	          Ok(self.evaluate_assignment(assignment)?.to_owned())
   236	        } else {
   237	          Err(Error::Internal {
   238	            message: format!("attempted to evaluate undefined variable `{variable}`"),
   239	          })
   240	        }
   241	      }
   242	    }
   243	  }

   245	  fn evaluate_condition(&mut self, condition: &Condition<'src>) -> RunResult<'src, bool> {
   246	    let lhs_value = self.evaluate_expression(&condition.lhs)?;
   247	    let rhs_value = self.evaluate_expression(&condition.rhs)?;
   248	    let condition = match condition.operator {
   249	      ConditionalOperator::Equality => lhs_value == rhs_value,
   250	      ConditionalOperator::Inequality => lhs_value != rhs_value,
   251	      ConditionalOperator::RegexMatch => Regex::new(&rhs_value)
   252	        .map_err(|source| Error::RegexCompile { source })?
   253	        .is_match(&lhs_value),
   254	      ConditionalOperator::RegexMismatch => !Regex::new(&rhs_value)
   255	        .map_err(|source| Error::RegexCompile { source })?
   256	        .is_match(&lhs_value),
   257	    };
   258	    Ok(condition)

   298	  pub(crate) fn evaluate_line(
   299	    &mut self,
   300	    line: &Line<'src>,
   301	    continued: bool,
   302	  ) -> RunResult<'src, String> {
   303	    let mut evaluated = String::new();
   304	    for (i, fragment) in line.fragments.iter().enumerate() {
   305	      match fragment {
   306	        Fragment::Text { token } => {
   307	          let lexeme = token.lexeme().replace("{{{{", "{{");
   308	
   309	          if i == 0 && continued {
   310	            evaluated += lexeme.trim_start();
   311	          } else {
   312	            evaluated += &lexeme;
   313	          }
   314	        }
   315	        Fragment::Interpolation { expression } => {
   316	          evaluated += &self.evaluate_expression(expression)?;
   317	        }
   318	      }
   319	    }
   320	    Ok(evaluated)
   321	  }

   323	  pub(crate) fn evaluate_parameters(
   324	    context: &ExecutionContext<'src, 'run>,
   325	    is_dependency: bool,
   326	    arguments: &[String],
   327	    parameters: &[Parameter<'src>],
   328	  ) -> RunResult<'src, (Scope<'src, 'run>, Vec<String>)> {
   329	    let mut evaluator = Self::new(context, is_dependency, context.scope);
   330	
   331	    let mut positional = Vec::new();
   332	
   333	    let mut rest = arguments;
   334	    for parameter in parameters {
   335	      let value = if rest.is_empty() {
   336	        if let Some(ref default) = parameter.default {
   337	          let value = evaluator.evaluate_expression(default)?;
   338	          positional.push(value.clone());
   339	          value
   340	        } else if parameter.kind == ParameterKind::Star {
   341	          String::new()
   342	        } else {
   343	          return Err(Error::Internal {
   344	            message: "missing parameter without default".to_owned(),
   345	          });
   346	        }
   347	      } else if parameter.kind.is_variadic() {
   348	        for value in rest {
   349	          positional.push(value.clone());
   350	        }
   351	        let value = rest.to_vec().join(" ");
   352	        rest = &[];
   353	        value
   354	      } else {
   355	        let value = rest[0].clone();
   356	        positional.push(value.clone());
   357	        rest = &rest[1..];
   358	        value
   359	      };
   360	      evaluator.scope.bind(Binding {
   361	        constant: false,
   362	        export: parameter.export,
   363	        file_depth: 0,
   364	        name: parameter.name,
   365	        private: false,
   366	        value,
   367	      });
   368	    }
   369	
   370	    Ok((evaluator.scope, positional))
   371	  }
