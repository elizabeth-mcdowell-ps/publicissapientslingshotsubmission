   232	   * to a fixed-point number between 0 and 1, representing that item's proportion of traffic (1 ==

   234	   * special case a single item, and so that we can round up very low values without risking uint32
   235	   * overflow of the sum of weights.

   241	    long one = 1L << FIXED_POINT_FRACTIONAL_BITS;

   419	    private SocketAddress rewriteAddress(SocketAddress addr,
   420	        ImmutableMap<String, Object> endpointMetadata,
   421	        ImmutableMap<String, Object> localityMetadata) {
   422	      if (!(addr instanceof InetSocketAddress)) {
   423	        return addr;
   424	      }
   425	
   426	      SocketAddress proxyAddress;
   427	      try {
   428	        proxyAddress = (SocketAddress) endpointMetadata.get(
   429	            "envoy.http11_proxy_transport_socket.proxy_address");
   430	        if (proxyAddress == null) {
   431	          proxyAddress = (SocketAddress) localityMetadata.get(
   432	              "envoy.http11_proxy_transport_socket.proxy_address");
   433	        }
   434	      } catch (ClassCastException e) {
   435	        return addr;
   436	      }
   437	
   438	      if (proxyAddress == null) {
   439	        return addr;
   440	      }
   441	
   442	      return HttpConnectProxiedSocketAddress.newBuilder()
   443	          .setTargetAddress((InetSocketAddress) addr)
   444	          .setProxyAddress(proxyAddress)
   445	          .build();
   446	    }

   448	    private List<String> generatePriorityNames(String name,
   449	        Map<Locality, LocalityLbEndpoints> localityLbEndpoints) {
   450	      TreeMap<Integer, List<Locality>> todo = new TreeMap<>();
   451	      for (Locality locality : localityLbEndpoints.keySet()) {
   452	        int priority = localityLbEndpoints.get(locality).priority();
   453	        if (!todo.containsKey(priority)) {
   454	          todo.put(priority, new ArrayList<>());
   455	        }
   456	        todo.get(priority).add(locality);
   457	      }
   458	      Map<Locality, String> newNames = new HashMap<>();
   459	      Set<String> usedNames = new HashSet<>();
   460	      List<String> ret = new ArrayList<>();
   461	      for (Integer priority: todo.keySet()) {
   462	        String foundName = "";
   463	        for (Locality locality : todo.get(priority)) {
   464	          if (localityPriorityNames.containsKey(locality)
   465	              && usedNames.add(localityPriorityNames.get(locality))) {
   466	            foundName = localityPriorityNames.get(locality);
   467	            break;
   468	          }
   469	        }
   470	        if ("".equals(foundName)) {
   471	          foundName = priorityName(name, priorityNameGenId++);
   472	        }
   473	        for (Locality locality : todo.get(priority)) {
   474	          newNames.put(locality, foundName);
   475	        }
   476	        ret.add(foundName);
   477	      }
   478	      localityPriorityNames = newNames;
   479	      return ret;
   480	    }

   500	   * Generates configs to be used in the priority LB policy for priorities in a cluster.
   501	   *
   502	   * <p>priority LB -> cluster_impl LB (one per priority) -> (weighted_target LB
   503	   * -> round_robin / least_request_experimental (one per locality)) / ring_hash_experimental

   505	  private static Map<String, PriorityChildConfig> generatePriorityChildConfigs(
   506	      String clusterName,
   507	      CdsUpdate discovery,
   508	      Object endpointLbConfig,
   509	      LoadBalancerRegistry lbRegistry,
   510	      Map<String, Map<Locality, Integer>> prioritizedLocalityWeights,
   511	      List<DropOverload> dropOverloads) {
   512	    Map<String, PriorityChildConfig> configs = new HashMap<>();
   513	    for (String priority : prioritizedLocalityWeights.keySet()) {
   514	      ClusterImplConfig clusterImplConfig =
   515	          new ClusterImplConfig(
   516	              clusterName, discovery.edsServiceName(), discovery.lrsServerInfo(),
   517	              discovery.maxConcurrentRequests(), dropOverloads, endpointLbConfig,
   518	              discovery.upstreamTlsContext(), discovery.filterMetadata(),
   519	              discovery.backendMetricPropagation());
   520	      LoadBalancerProvider clusterImplLbProvider =
   521	          lbRegistry.getProvider(XdsLbPolicies.CLUSTER_IMPL_POLICY_NAME);
   522	      Object priorityChildPolicy = GracefulSwitchLoadBalancer.createLoadBalancingPolicyConfig(
   523	          clusterImplLbProvider, clusterImplConfig);
   524	
   525	      // If outlier detection has been configured we wrap the child policy in the outlier detection
   526	      // load balancer.

   528	        LoadBalancerProvider outlierDetectionProvider = lbRegistry.getProvider(
   529	            "outlier_detection_experimental");

   531	            outlierDetectionProvider,

   537	          new PriorityChildConfig(priorityChildPolicy, isEds /* ignoreReresolution */);
   538	      configs.put(priority, priorityChildConfig);

    72	/**
    73	 * Load balancer for cluster_impl_experimental LB policy. This LB policy is the child LB policy of
    74	 * the priority_experimental LB policy and the parent LB policy of the weighted_target_experimental
    75	 * LB policy in the xDS load balancing hierarchy. This LB policy applies cluster-level
    76	 * configurations to requests sent to the corresponding cluster, such as drop policies, circuit
    77	 * breakers.

   119	  @Override
   120	  public Status acceptResolvedAddresses(ResolvedAddresses resolvedAddresses) {
   121	    logger.log(XdsLogLevel.DEBUG, "Received resolution result: {0}", resolvedAddresses);
   122	    Attributes attributes = resolvedAddresses.getAttributes();
   123	    if (xdsClient == null) {
   124	      xdsClient = checkNotNull(attributes.get(io.grpc.xds.XdsAttributes.XDS_CLIENT), "xdsClient");
   125	    }
   126	    if (callCounterProvider == null) {
   127	      callCounterProvider = attributes.get(io.grpc.xds.XdsAttributes.CALL_COUNTER_PROVIDER);
   128	    }
   129	
   130	    ClusterImplConfig config =
   131	        (ClusterImplConfig) resolvedAddresses.getLoadBalancingPolicyConfig();
   132	    if (config == null) {
   133	      return Status.INTERNAL.withDescription("No cluster configuration found");
   134	    }
   135	
   136	    if (cluster == null) {
   137	      cluster = config.cluster;
   138	      edsServiceName = config.edsServiceName;
   139	      childLbHelper = new ClusterImplLbHelper(
   140	          callCounterProvider.getOrCreate(config.cluster, config.edsServiceName),
   141	          config.lrsServerInfo);
   142	      childSwitchLb = new GracefulSwitchLoadBalancer(childLbHelper);
   143	      // Assume load report server does not change throughout cluster lifetime.
   144	      if (config.lrsServerInfo != null) {
   145	        dropStats = xdsClient.addClusterDropStats(config.lrsServerInfo, cluster, edsServiceName);
   146	      }
   147	    }
   148	
   149	    childLbHelper.updateDropPolicies(config.dropCategories);
   150	    childLbHelper.updateMaxConcurrentRequests(config.maxConcurrentRequests);
   151	    childLbHelper.updateSslContextProviderSupplier(config.tlsContext);
   152	    childLbHelper.updateFilterMetadata(config.filterMetadata);
   153	    childLbHelper.updateBackendMetricPropagation(config.backendMetricPropagation);
   154	
   155	    return childSwitchLb.acceptResolvedAddresses(
   156	        resolvedAddresses.toBuilder()
   157	            .setAttributes(attributes.toBuilder()
   158	              .set(NameResolver.ATTR_BACKEND_SERVICE, cluster)
   159	              .build())
   160	            .setLoadBalancingPolicyConfig(config.childConfig)
   161	            .build());
   162	  }

   164	  @Override
   165	  public void handleNameResolutionError(Status error) {
   166	    if (childSwitchLb != null) {
   167	      childSwitchLb.handleNameResolutionError(error);
   168	    } else {
   169	      helper.updateBalancingState(
   170	          ConnectivityState.TRANSIENT_FAILURE, new FixedResultPicker(PickResult.withError(error)));
   171	    }
   172	  }

   174	  @Override
   175	  public void requestConnection() {
   176	    if (childSwitchLb != null) {
   177	      childSwitchLb.requestConnection();
   178	    }
   179	  }

   181	  @Override
   182	  public void shutdown() {
   183	    if (dropStats != null) {
   184	      dropStats.release();
   185	    }
   186	    if (childSwitchLb != null) {
   187	      childSwitchLb.shutdown();
   188	      if (childLbHelper != null) {
   189	        childLbHelper.updateSslContextProviderSupplier(null);
   190	        childLbHelper = null;
   191	      }
   192	    }
   193	    xdsClient = null;
   194	  }

   220	    public void updateBalancingState(ConnectivityState newState, SubchannelPicker newPicker) {
   221	      currentState = newState;
   222	      currentPicker =  newPicker;
   223	      SubchannelPicker picker = new RequestLimitingSubchannelPicker(

   228	    @Override
   229	    public Subchannel createSubchannel(CreateSubchannelArgs args) {
   230	      List<EquivalentAddressGroup> addresses = withAdditionalAttributes(args.getAddresses());
   231	      // This value for  ClusterLocality is not recommended for general use.
   232	      // Currently, we extract locality data from the first address, even before the subchannel is
   233	      // READY.
   234	      // This is mainly to accommodate scenarios where a Load Balancing API (like "pick first")
   235	      // might return the subchannel before it is READY. Typically, we wouldn't report load for such
   236	      // selections because the channel will disregard the chosen (not-ready) subchannel.
   237	      // However, we needed to ensure this case is handled.
   238	      ClusterLocality clusterLocality = createClusterLocalityFromAttributes(
   239	          args.getAddresses().get(0).getAttributes());
   240	      AtomicReference<ClusterLocality> localityAtomicReference = new AtomicReference<>(
   241	          clusterLocality);
   242	      Attributes.Builder attrsBuilder = args.getAttributes().toBuilder()
   243	          .set(ATTR_CLUSTER_LOCALITY, localityAtomicReference);
   244	      if (GrpcUtil.getFlag("GRPC_EXPERIMENTAL_XDS_AUTHORITY_REWRITE", false)) {
   245	        String hostname = args.getAddresses().get(0).getAttributes()
   246	            .get(XdsInternalAttributes.ATTR_ADDRESS_NAME);
   247	        if (hostname != null) {
   248	          attrsBuilder.set(ATTR_SUBCHANNEL_ADDRESS_NAME, hostname);
   249	        }
   250	      }
   251	      args = args.toBuilder().setAddresses(addresses).setAttributes(attrsBuilder.build()).build();
   252	      final Subchannel subchannel = delegate().createSubchannel(args);
   253	
   254	      return new ClusterImplSubchannel(subchannel, localityAtomicReference);
   255	    }

   257	    private final class ClusterImplSubchannel extends ForwardingSubchannel {
   258	      private final Subchannel delegate;
   259	      private final AtomicReference<ClusterLocality> localityAtomicReference;
   260	
   261	      private ClusterImplSubchannel(
   262	          Subchannel delegate, AtomicReference<ClusterLocality> localityAtomicReference) {
   263	        this.delegate = delegate;
   264	        this.localityAtomicReference = localityAtomicReference;
   265	      }
   266	
   267	      @Override
   268	      public void start(SubchannelStateListener listener) {
   269	        delegate().start(
   270	            new SubchannelStateListener() {
   271	              @Override
   272	              public void onSubchannelState(ConnectivityStateInfo newState) {
   273	                // Do nothing if LB has been shutdown
   274	                if (xdsClient != null && newState.getState().equals(ConnectivityState.READY)) {
   275	                  // Get locality based on the connected address attributes
   276	                  ClusterLocality updatedClusterLocality =
   277	                      createClusterLocalityFromAttributes(
   278	                          delegate.getConnectedAddressAttributes());
   279	                  ClusterLocality oldClusterLocality =
   280	                      localityAtomicReference.getAndSet(updatedClusterLocality);
   281	                  oldClusterLocality.release();
   282	                }
   283	                listener.onSubchannelState(newState);
   284	              }
   285	            });
   286	      }
   287	
   288	      @Override
   289	      public void shutdown() {
   290	        localityAtomicReference.get().release();
   291	        delegate().shutdown();
   292	      }

   305	    private List<EquivalentAddressGroup> withAdditionalAttributes(
   306	        List<EquivalentAddressGroup> addresses) {
   307	      List<EquivalentAddressGroup> newAddresses = new ArrayList<>();
   308	      for (EquivalentAddressGroup eag : addresses) {
   309	        Attributes.Builder attrBuilder = eag.getAttributes().toBuilder().set(
   310	            io.grpc.xds.XdsAttributes.ATTR_CLUSTER_NAME, cluster);
   311	        if (sslContextProviderSupplier != null) {
   312	          attrBuilder.set(
   313	              SecurityProtocolNegotiators.ATTR_SSL_CONTEXT_PROVIDER_SUPPLIER,
   314	              sslContextProviderSupplier);
   315	        }
   316	        newAddresses.add(new EquivalentAddressGroup(eag.getAddresses(), attrBuilder.build()));
   317	      }
   318	      return newAddresses;
   319	    }

   321	    private ClusterLocality createClusterLocalityFromAttributes(Attributes addressAttributes) {
   322	      Locality locality = addressAttributes.get(io.grpc.xds.XdsAttributes.ATTR_LOCALITY);
   323	      String localityName = addressAttributes.get(EquivalentAddressGroup.ATTR_LOCALITY_NAME);
   324	
   325	      // Endpoint addresses resolved by ClusterResolverLoadBalancer should always contain
   326	      // attributes with its locality, including endpoints in LOGICAL_DNS clusters.
   327	      // In case of not (which really shouldn't), loads are aggregated under an empty
   328	      // locality.
   329	      if (locality == null) {
   330	        locality = Locality.create("", "", "");
   331	        localityName = "";
   332	      }
   333	
   334	      final ClusterLocalityStats localityStats =
   335	          (lrsServerInfo == null)
   336	              ? null
   337	              : xdsClient.addClusterLocalityStats(lrsServerInfo, cluster,
   338	                  edsServiceName, locality, backendMetricPropagation);
   339	
   340	      return new ClusterLocality(localityStats, localityName);
   341	    }

   355	    private void updateMaxConcurrentRequests(@Nullable Long maxConcurrentRequests) {
   356	      if (Objects.equals(this.maxConcurrentRequests, maxConcurrentRequests)) {
   357	        return;
   358	      }
   359	      this.maxConcurrentRequests =
   360	          maxConcurrentRequests != null
   361	              ? maxConcurrentRequests
   362	              : DEFAULT_PER_CLUSTER_MAX_CONCURRENT_REQUESTS;
   363	      updateBalancingState(currentState, currentPicker);

   366	    private void updateSslContextProviderSupplier(@Nullable UpstreamTlsContext tlsContext) {
   367	      UpstreamTlsContext currentTlsContext =
   368	          sslContextProviderSupplier != null
   369	              ? (UpstreamTlsContext)sslContextProviderSupplier.getTlsContext()
   370	              : null;
   371	      if (Objects.equals(currentTlsContext,  tlsContext)) {
   372	        return;
   373	      }
   374	      if (sslContextProviderSupplier != null) {
   375	        sslContextProviderSupplier.close();
   376	      }
   377	      sslContextProviderSupplier =
   378	          tlsContext != null
   379	              ? new SslContextProviderSupplier(tlsContext,
   380	                                               (TlsContextManager) xdsClient.getSecurityConfig())
   381	              : null;
   382	    }

   393	    private class RequestLimitingSubchannelPicker extends SubchannelPicker {
   394	      private final SubchannelPicker delegate;
   395	      private final List<DropOverload> dropPolicies;
   396	      private final long maxConcurrentRequests;
   397	      private final Map<String, Struct> filterMetadata;
   398	
   399	      private RequestLimitingSubchannelPicker(SubchannelPicker delegate,
   400	          List<DropOverload> dropPolicies, long maxConcurrentRequests,
   401	          Map<String, Struct> filterMetadata) {
   402	        this.delegate = delegate;
   403	        this.dropPolicies = dropPolicies;
   404	        this.maxConcurrentRequests = maxConcurrentRequests;
   405	        this.filterMetadata = checkNotNull(filterMetadata, "filterMetadata");
   406	      }
   407	
   408	      @Override
   409	      public PickResult pickSubchannel(PickSubchannelArgs args) {
   410	        args.getCallOptions().getOption(ClusterImplLoadBalancerProvider.FILTER_METADATA_CONSUMER)
   411	            .accept(filterMetadata);
   412	        args.getPickDetailsConsumer().addOptionalLabel("grpc.lb.backend_service", cluster);
   413	        for (DropOverload dropOverload : dropPolicies) {
   414	          int rand = random.nextInt(1_000_000);
   415	          if (rand < dropOverload.dropsPerMillion()) {
   416	            logger.log(XdsLogLevel.INFO, "Drop request with category: {0}",
   417	                dropOverload.category());
   418	            if (dropStats != null) {
   419	              dropStats.recordDroppedRequest(dropOverload.category());
   420	            }
   421	            return PickResult.withDrop(
   422	                Status.UNAVAILABLE.withDescription("Dropped: " + dropOverload.category()));
   423	          }
   424	        }
   425	        PickResult result = delegate.pickSubchannel(args);
   426	        if (result.getStatus().isOk() && result.getSubchannel() != null) {
   427	          Subchannel subchannel = result.getSubchannel();
   428	          if (subchannel instanceof ClusterImplLbHelper.ClusterImplSubchannel) {
   429	            subchannel = ((ClusterImplLbHelper.ClusterImplSubchannel) subchannel).delegate();
   430	            result = result.copyWithSubchannel(subchannel);
   431	          }
   432	          if (enableCircuitBreaking) {
   433	            if (inFlights.get() >= maxConcurrentRequests) {
   434	              if (dropStats != null) {
   435	                dropStats.recordDroppedRequest();
   436	              }
   437	              return PickResult.withDrop(Status.UNAVAILABLE.withDescription(
   438	                  String.format(Locale.US, "Cluster max concurrent requests limit of %d exceeded",
   439	                      maxConcurrentRequests)));
   440	            }
   441	          }
   442	          final AtomicReference<ClusterLocality> clusterLocality =
   443	              result.getSubchannel().getAttributes().get(ATTR_CLUSTER_LOCALITY);
   444	
   445	          if (clusterLocality != null) {
   446	            ClusterLocalityStats stats = clusterLocality.get().getClusterLocalityStats();
   447	            if (stats != null) {
   448	              String localityName =
   449	                  result.getSubchannel().getAttributes().get(ATTR_CLUSTER_LOCALITY).get()
   450	                      .getClusterLocalityName();
   451	              args.getPickDetailsConsumer().addOptionalLabel("grpc.lb.locality", localityName);
   452	
   453	              ClientStreamTracer.Factory tracerFactory = new CountingStreamTracerFactory(
   454	                  stats, inFlights, result.getStreamTracerFactory());
   455	              ClientStreamTracer.Factory orcaTracerFactory = OrcaPerRequestUtil.getInstance()
   456	                  .newOrcaClientStreamTracerFactory(tracerFactory, new OrcaPerRpcListener(stats));
   457	              result = result.copyWithStreamTracerFactory(orcaTracerFactory);
   458	            }
   459	          }
   460	          if (args.getCallOptions().getOption(XdsNameResolver.AUTO_HOST_REWRITE_KEY) != null
   461	              && args.getCallOptions().getOption(XdsNameResolver.AUTO_HOST_REWRITE_KEY)) {
   462	            result = PickResult.withSubchannel(result.getSubchannel(),
   463	                result.getStreamTracerFactory(),
   464	                result.getSubchannel().getAttributes().get(ATTR_SUBCHANNEL_ADDRESS_NAME));
   465	          }
   466	        }
   467	        return result;
   468	      }

   477	  private static final class CountingStreamTracerFactory extends
   478	      ClientStreamTracer.Factory {
   479	    private final ClusterLocalityStats stats;
   480	    private final AtomicLong inFlights;
   481	    @Nullable
   482	    private final ClientStreamTracer.Factory delegate;
   483	
   484	    private CountingStreamTracerFactory(
   485	        ClusterLocalityStats stats, AtomicLong inFlights,
   486	        @Nullable ClientStreamTracer.Factory delegate) {
   487	      this.stats = checkNotNull(stats, "stats");
   488	      this.inFlights = checkNotNull(inFlights, "inFlights");
   489	      this.delegate = delegate;
   490	    }
   491	
   492	    @Override
   493	    public ClientStreamTracer newClientStreamTracer(StreamInfo info, Metadata headers) {
   494	      stats.recordCallStarted();
   495	      inFlights.incrementAndGet();
   496	      if (delegate == null) {
   497	        return new ClientStreamTracer() {
   498	          @Override
   499	          public void streamClosed(Status status) {
   500	            stats.recordCallFinished(status);
   501	            inFlights.decrementAndGet();
   502	          }
   503	        };
   504	      }
   505	      final ClientStreamTracer delegatedTracer = delegate.newClientStreamTracer(info, headers);
   506	      return new ForwardingClientStreamTracer() {
   507	        @Override
   508	        protected ClientStreamTracer delegate() {
   509	          return delegatedTracer;
   510	        }
   511	
   512	        @Override
   513	        public void streamClosed(Status status) {
   514	          stats.recordCallFinished(status);
   515	          inFlights.decrementAndGet();
   516	          delegate().streamClosed(status);
   517	        }
   518	      };
   519	    }

   522	  private static final class OrcaPerRpcListener implements OrcaPerRequestReportListener {
   523	
   524	    private final ClusterLocalityStats stats;
   525	
   526	    private OrcaPerRpcListener(ClusterLocalityStats stats) {
   527	      this.stats = checkNotNull(stats, "stats");
   528	    }
   529	
   530	    /**
   531	     * Copies ORCA metrics from {@link MetricReport} to {@link ClusterLocalityStats}
   532	     * such that they are included in the snapshot for the LRS report sent to the LRS server.
   533	     * This includes both top-level metrics (CPU, memory, application utilization) and named
   534	     * metrics, filtered according to the backend metric propagation configuration.
   535	     */
   536	    @Override
   537	    public void onLoadReport(MetricReport report) {
   538	      if (isEnabledOrcaLrsPropagation) {
   539	        stats.recordTopLevelMetrics(
   540	            report.getCpuUtilization(),
   541	            report.getMemoryUtilization(),
   542	            report.getApplicationUtilization());
   543	      }
   544	      stats.recordBackendLoadMetricStats(report.getNamedMetrics());
   545	    }
   546	  }
