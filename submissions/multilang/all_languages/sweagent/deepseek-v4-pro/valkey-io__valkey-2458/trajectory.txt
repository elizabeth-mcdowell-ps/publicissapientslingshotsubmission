    61	static int slotStatForSortAscCmp(const void *a, const void *b) {
    62	    slotStatForSort entry_a = *((slotStatForSort *)a);
    63	    slotStatForSort entry_b = *((slotStatForSort *)b);
    64	    if (entry_a.stat == entry_b.stat) {

    66	    }

    71	static int slotStatForSortDescCmp(const void *a, const void *b) {
    72	    slotStatForSort entry_a = *((slotStatForSort *)a);
    73	    slotStatForSort entry_b = *((slotStatForSort *)b);
    74	    if (entry_b.stat == entry_a.stat) {
    75	        return entry_a.slot - entry_b.slot;
    76	    }

    81	    int i = 0;

   147	/* Accumulates egress bytes upon sending replication stream. This only applies for primary nodes. */
   148	static void clusterSlotStatsUpdateNetworkBytesOutForReplication(long long len) {
   149	    client *c = server.current_client;
   150	    if (c == NULL || !clusterSlotStatsEnabled(c->slot)) return;
   151	
   152	    /* We multiply the bytes len by the number of replicas to account for us broadcasting to multiple replicas at once. */
   153	    len *= (long long)listLength(server.replicas);
   154	    serverAssert(c->slot >= 0 && c->slot < CLUSTER_SLOTS);
   155	    serverAssert(nodeIsPrimary(server.cluster->myself));
   156	    /* We sometimes want to adjust the counter downwards (for example when we want to undo accounting for
   157	     * SELECT commands that don't belong to any slot) so let's make sure we don't underflow the counter. */
   158	    serverAssert(len >= 0 || server.cluster->slot_stats[c->slot].network_bytes_out >= (uint64_t)-len);
   159	    server.cluster->slot_stats[c->slot].network_bytes_out += len;
   160	}
