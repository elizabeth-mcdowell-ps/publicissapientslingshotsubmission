   207	function isModelKnown(modelId: string): boolean {
   208	  const bareId = modelId.includes("/") ? modelId.split("/").pop()! : modelId;
   209	
   210	  if (MODEL_CAPABILITY_TIER[bareId]) return true;
   211	
   212	  for (const knownId of Object.keys(MODEL_CAPABILITY_TIER)) {
   213	    if (bareId.includes(knownId) || knownId.includes(bareId)) return true;
   214	  }
   215	
   216	  return false;
   217	}

   219	function getModelTier(modelId: string): ComplexityTier {
   220	  // Strip provider prefix if present
   221	  const bareId = modelId.includes("/") ? modelId.split("/").pop()! : modelId;
   222	
   223	  // Check exact match first
   224	  if (MODEL_CAPABILITY_TIER[bareId]) return MODEL_CAPABILITY_TIER[bareId];
   225	
   226	  // Check if any known model ID is a prefix/suffix match
   227	  for (const [knownId, tier] of Object.entries(MODEL_CAPABILITY_TIER)) {
   228	    if (bareId.includes(knownId) || knownId.includes(bareId)) return tier;
   229	  }
   230	
   231	  // Unknown models are assumed heavy (safest assumption)
   232	  return "heavy";
   233	}
   234	
   235	function findModelForTier(
   236	  tier: ComplexityTier,
   237	  config: DynamicRoutingConfig,
   238	  availableModelIds: string[],
   239	  crossProvider: boolean,
   240	): string | null {
   241	  // 1. Check explicit tier_models config
   242	  const explicitModel = config.tier_models?.[tier];
   243	  if (explicitModel && availableModelIds.includes(explicitModel)) {
   244	    return explicitModel;
   245	  }
   246	  // Also check with provider prefix stripped
   247	  if (explicitModel) {
   248	    const match = availableModelIds.find(id => {
   249	      const bareAvail = id.includes("/") ? id.split("/").pop()! : id;
   250	      const bareExplicit = explicitModel.includes("/") ? explicitModel.split("/").pop()! : explicitModel;
   251	      return bareAvail === bareExplicit;
   252	    });
   253	    if (match) return match;
   254	  }
   255	
   256	  // 2. Auto-detect: find the cheapest available model in the requested tier
   257	  const candidates = availableModelIds
   258	    .filter(id => {
   259	      const modelTier = getModelTier(id);
   260	      return modelTier === tier;
   261	    })
   262	    .sort((a, b) => {
   263	      if (!crossProvider) return 0;
   264	      const costA = getModelCost(a);
   265	      const costB = getModelCost(b);
   266	      return costA - costB;
   267	    });
   268	
   269	  return candidates[0] ?? null;
   270	}
   271	
   272	function getModelCost(modelId: string): number {
   273	  const bareId = modelId.includes("/") ? modelId.split("/").pop()! : modelId;
   274	
   275	  if (MODEL_COST_PER_1K_INPUT[bareId] !== undefined) {
   276	    return MODEL_COST_PER_1K_INPUT[bareId];
   277	  }
   278	
   279	  // Check partial matches
   280	  for (const [knownId, cost] of Object.entries(MODEL_COST_PER_1K_INPUT)) {
   281	    if (bareId.includes(knownId) || knownId.includes(bareId)) return cost;
   282	  }
   283	
   284	  // Unknown cost — assume expensive to avoid routing to unknown cheap models
   285	  return 999;
   286	}
