   275	    private String resolveService(String authorization, String action) {
   276	        if (authorization != null && !authorization.isEmpty()) {
   277	            Matcher m = SERVICE_PATTERN.matcher(authorization);
   278	            if (m.find()) {

   280	                ServiceDescriptor descriptor = catalog.byCredentialScope(scope).orElse(null);
   281	                if (descriptor != null && descriptor.supportsProtocol(ServiceProtocol.QUERY)) {
   282	                    return descriptor.externalKey();

   284	            }
