   253	    ///
   254	    /// Specifically, returns:
   255	    /// - the type of the variable,
   256	    /// - the kind of assignment target (`Local`, `Storage`, `ExternalStorage`, or `Err`).
   257	    ///
   258	    /// This also performs static checks for:
   259	    /// - assignments to constants,
   260	    /// - assignments to storage vectors (invalid),

   262	    /// - assignments to variables outside an async block’s scope.
   263	    pub fn visit_path_assign(&mut self, input: &Path) -> AssignTargetInfo {
   264	        let current_program = self.scope_state.program_name.unwrap();
   265	
   266	        // Lookup the variable in the symbol table
   267	        let Some(var) = self.state.symbol_table.lookup_path(current_program, input) else {
   268	            self.emit_err(TypeCheckerError::unknown_sym("variable", input, input.span));
   269	            return AssignTargetInfo { ty: Type::Err, kind: AssignTargetKind::Err };
   270	        };
   271	
   272	        let ty = var.type_.expect("must be known by now").clone();
   273	
   274	        // Cannot assign to storage vectors
   275	        if ty.is_vector() {
   276	            self.emit_err(TypeCheckerError::invalid_assignment_target(input, input.span()));
   277	            return AssignTargetInfo { ty: Type::Err, kind: AssignTargetKind::Err };
   278	        }
   279	
   280	        // Check that variable is not a constant or constant input

   283	            VariableType::ConstParameter => {

   288	            }
   289	            VariableType::Storage => {
   290	                // Determine if this storage variable belongs to another program
   291	                let kind = if input.user_program().is_some() {
   292	                    AssignTargetKind::ExternalStorage
   293	                } else {
   294	                    AssignTargetKind::Storage
   295	                };

   297	            }
   298	            VariableType::Mut | VariableType::Input(_) => {}
   299	        }
   300	
   301	        // Ensure assignment is allowed in async function conditional scopes
   302	        if self.scope_state.variant.unwrap().is_onchain() && !self.symbol_in_conditional_scope(input.identifier().name)
   303	        {
   304	            self.emit_err(TypeCheckerError::final_cannot_assign_outside_conditional(input, "fn", var.span));
   305	        }

   318	                input.span,
   319	            ));
   320	        }
   321	
   322	        // Default: assignment to a local variable
   323	        AssignTargetInfo { ty, kind: AssignTargetKind::Local }
   324	    }
