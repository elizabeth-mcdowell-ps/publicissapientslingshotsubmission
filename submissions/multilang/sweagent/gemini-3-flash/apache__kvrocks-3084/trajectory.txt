   434	    GetFixed32(input, &ef_runtime);
   435	    GetDouble(input, &epsilon);
   436	    GetFixed16(input, &num_levels);
   437	    return rocksdb::Status::OK();
   438	  }
   439	};
   440	
   441	struct HnswNodeFieldMetadata {
   442	  uint16_t num_neighbours;
   443	  std::vector<double> vector;
   444	
   445	  HnswNodeFieldMetadata() = default;
   446	  HnswNodeFieldMetadata(uint16_t num_neighbours, std::vector<double> vector)
   447	      : num_neighbours(num_neighbours), vector(std::move(vector)) {}
   448	
   449	  void Encode(std::string *dst) const {
   450	    PutFixed16(dst, num_neighbours);
   451	    PutFixed16(dst, static_cast<uint16_t>(vector.size()));
   452	    for (double element : vector) {
   453	      PutDouble(dst, element);
   454	    }
   455	  }
   456	
   457	  rocksdb::Status Decode(Slice *input) {
   458	    if (input->size() < 2 + 2) {
   459	      return rocksdb::Status::Corruption(kErrorInsufficientLength);
   460	    }
   461	    GetFixed16(input, (uint16_t *)(&num_neighbours));
   462	
   463	    uint16_t dim = 0;
   464	    GetFixed16(input, (uint16_t *)(&dim));
   465	
   466	    if (input->size() != dim * sizeof(double)) {
   467	      return rocksdb::Status::Corruption(kErrorIncorrectLength);
   468	    }
   469	    vector.resize(dim);
   470	
   471	    for (auto i = 0; i < dim; ++i) {
   472	      GetDouble(input, &vector[i]);
   473	    }
   474	    return rocksdb::Status::OK();
   475	  }
   476	};

   478	inline rocksdb::Status IndexFieldMetadata::Decode(Slice *input, std::unique_ptr<IndexFieldMetadata> &ptr) {
   479	  if (input->size() < 1) {
   480	    return rocksdb::Status::Corruption(kErrorInsufficientLength);
   481	  }
   482	
   483	  switch (DecodeType((*input)[0])) {
   484	    case IndexFieldType::TAG:
   485	      ptr = std::make_unique<TagFieldMetadata>();
   486	      break;
   487	    case IndexFieldType::NUMERIC:
   488	      ptr = std::make_unique<NumericFieldMetadata>();
   489	      break;
   490	    case IndexFieldType::VECTOR:
   491	      ptr = std::make_unique<HnswVectorFieldMetadata>();
   492	      break;
   493	    default:
   494	      return rocksdb::Status::Corruption("encountered unknown field type");
   495	  }
   496	
   497	  return ptr->Decode(input);
   498	}

    49	Status SubKeyFilter::GetMetadata(const InternalKey &ikey, Metadata *metadata) const {
    50	  auto db = stor_->GetDB();
    51	  const auto cf_handles = stor_->GetCFHandles();
    52	  // storage close the would delete the column family handler and DB
    53	  if (!db || cf_handles->size() < 2) return {Status::NotOK, "storage is closed"};
    54	  std::string metadata_key = ComposeNamespaceKey(ikey.GetNamespace(), ikey.GetKey(), stor_->IsSlotIdEncoded());
    55	
    56	  if (cached_key_.empty() || metadata_key != cached_key_) {
    57	    std::string bytes;
    58	    rocksdb::Status s = db->Get(rocksdb::ReadOptions(), (*cf_handles)[1], metadata_key, &bytes);
    59	    cached_key_ = std::move(metadata_key);
    60	    if (s.ok()) {
    61	      cached_metadata_ = std::move(bytes);
    62	    } else if (s.IsNotFound()) {
    63	      // metadata was deleted(perhaps compaction or manual)
    64	      // clear the metadata
    65	      cached_metadata_.clear();
    66	      return {Status::NotFound, "metadata is not found"};
    67	    } else {
    68	      cached_key_.clear();
    69	      cached_metadata_.clear();
    70	      return {Status::NotOK, "fetch error: " + s.ToString()};
    71	    }
    72	  }
    73	  // the metadata was not found
    74	  if (cached_metadata_.empty()) return {Status::NotFound, "metadata is not found"};
    75	  // the metadata is cached
    76	  rocksdb::Status s = metadata->Decode(cached_metadata_);
    77	  if (!s.ok()) {
    78	    cached_key_.clear();
    79	    return {Status::NotOK, "decode error: " + s.ToString()};

    84	bool SubKeyFilter::IsMetadataExpired(const InternalKey &ikey, const Metadata &metadata) {
    85	  // lazy delete to avoid race condition between command Expire and subkey Compaction
    86	  // Related issue:https://github.com/apache/kvrocks/issues/1298
    87	  //
    88	  // `util::GetTimeStampMS() - 300000` means extending 5 minutes for expired items,
    89	  // to prevent them from being recycled once they reach the expiration time.
    90	  uint64_t lazy_expired_ts = util::GetTimeStampMS() - 300000;
    91	  return metadata.IsSingleKVType()  // metadata key was overwrite by set command
    92	         || metadata.ExpireAt(lazy_expired_ts) || ikey.GetVersion() != metadata.version;
