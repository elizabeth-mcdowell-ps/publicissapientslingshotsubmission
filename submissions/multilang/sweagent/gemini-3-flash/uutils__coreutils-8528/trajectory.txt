   335	fn determine_overwrite_mode(matches: &ArgMatches) -> OverwriteMode {
   336	    // This does not exactly match the GNU implementation:
   337	    // The GNU mv defaults to Force, but if more than one of the
   338	    // overwrite options are supplied, only the last takes effect.
   339	    // To default to no-clobber in that situation seems safer:
   340	    //
   341	    if matches.get_flag(OPT_NO_CLOBBER) {
   342	        OverwriteMode::NoClobber
   343	    } else if matches.get_flag(OPT_INTERACTIVE) {
   344	        OverwriteMode::Interactive
   345	    } else {
   346	        OverwriteMode::Force
   347	    }
   348	}

   362	fn handle_two_paths(source: &Path, target: &Path, opts: &Options) -> UResult<()> {
   363	    if opts.backup == BackupMode::Simple && source_is_target_backup(source, target, &opts.suffix) {
   364	        return Err(io::Error::new(
   365	            io::ErrorKind::NotFound,
   366	            translate!("mv-error-backup-might-destroy-source", "target" => target.quote(), "source" => source.quote()),
   367	        )
   368	        .into());
   369	    }
   370	    if source.symlink_metadata().is_err() {
   371	        return Err(if path_ends_with_terminator(source) {
   372	            MvError::CannotStatNotADirectory(source.quote().to_string()).into()
   373	        } else {
   374	            MvError::NoSuchFile(source.quote().to_string()).into()
   375	        });
   376	    }
   377	
   378	    let target_is_dir = target.is_dir();
   379	    let source_is_dir = source.is_dir();
   380	
   381	    if path_ends_with_terminator(target)
   382	        && (!target_is_dir && !source_is_dir)
   383	        && !opts.no_target_dir
   384	        && opts.update != UpdateMode::IfOlder
   385	    {
   386	        return Err(MvError::FailedToAccessNotADirectory(target.quote().to_string()).into());
   387	    }
   388	
   389	    assert_not_same_file(source, target, target_is_dir, opts)?;
   390	
   391	    if target_is_dir {
   392	        if opts.no_target_dir {
   393	            if source.is_dir() {
   394	                #[cfg(unix)]
   395	                let (mut hardlink_tracker, hardlink_scanner) = create_hardlink_context();
   396	                #[cfg(unix)]
   397	                let hardlink_params = (Some(&mut hardlink_tracker), Some(&hardlink_scanner));
   398	                #[cfg(not(unix))]
   399	                let hardlink_params = (None, None);
   400	
   401	                rename(
   402	                    source,
   403	                    target,
   404	                    opts,
   405	                    None,
   406	                    hardlink_params.0,
   407	                    hardlink_params.1,
   408	                )
   409	                .map_err_context(|| {
   410	                    translate!("mv-error-cannot-move", "source" => source.quote(), "target" => target.quote())
   411	                })
   412	            } else {
   413	                Err(MvError::DirectoryToNonDirectory(target.quote().to_string()).into())
   414	            }
   415	        } else {
   416	            move_files_into_dir(&[source.to_path_buf()], target, opts)
   417	        }
   418	    } else if target.exists() && source.is_dir() {
   419	        match opts.overwrite {
   420	            OverwriteMode::NoClobber => return Ok(()),
   421	            OverwriteMode::Interactive => {
   422	                if !prompt_yes!(
   423	                    "{}",
   424	                    translate!("mv-prompt-overwrite", "target" => target.quote())
   425	                ) {
   426	                    return Err(io::Error::other("").into());
   427	                }
   428	            }
   429	            OverwriteMode::Force => {}
   430	        }
   431	        Err(MvError::NonDirectoryToDirectory(
   432	            source.quote().to_string(),
   433	            target.quote().to_string(),
   434	        )
   435	        .into())
   436	    } else {
   437	        #[cfg(unix)]
   438	        let (mut hardlink_tracker, hardlink_scanner) = create_hardlink_context();
   439	        #[cfg(unix)]
   440	        let hardlink_params = (Some(&mut hardlink_tracker), Some(&hardlink_scanner));
   441	        #[cfg(not(unix))]
   442	        let hardlink_params = (None, None);
   443	
   444	        rename(
   445	            source,
   446	            target,
   447	            opts,
   448	            None,
   449	            hardlink_params.0,
   450	            hardlink_params.1,
   451	        )
   452	        .map_err(|e| USimpleError::new(1, format!("{e}")))
   453	    }
   454	}
