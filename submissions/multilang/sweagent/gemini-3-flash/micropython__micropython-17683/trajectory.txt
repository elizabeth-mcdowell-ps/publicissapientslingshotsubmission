    92	static inline size_t proxy_c_add_obj(mp_obj_t obj) {
    93	    // Search for the first free slot in proxy_c_ref.
    94	    size_t id = 0;
    95	    mp_obj_list_t *l = (mp_obj_list_t *)MP_OBJ_TO_PTR(MP_STATE_PORT(proxy_c_ref));
    96	    while (proxy_c_ref_next < l->len) {
    97	        if (l->items[proxy_c_ref_next] == MP_OBJ_NULL) {
    98	            // Free slot found, reuse it.
    99	            id = proxy_c_ref_next;
   100	            ++proxy_c_ref_next;
   101	            l->items[id] = obj;
   102	            break;
   103	        }
   104	        ++proxy_c_ref_next;
   105	    }
   106	
   107	    if (id == 0) {
   108	        // No free slots, so grow proxy_c_ref by one (append at the end of the list).
   109	        id = l->len;
   110	        mp_obj_list_append(MP_STATE_PORT(proxy_c_ref), obj);
   111	        proxy_c_ref_next = l->len;
   112	    }
   113	
   114	    // Add the object to proxy_c_dict, keyed by the object pointer, with value the object id.
   115	    mp_obj_t obj_key = mp_obj_new_int_from_uint((uintptr_t)obj);
   116	    mp_map_elem_t *elem = mp_map_lookup(mp_obj_dict_get_map(MP_STATE_PORT(proxy_c_dict)), obj_key, MP_MAP_LOOKUP_ADD_IF_NOT_FOUND);
   117	    elem->value = mp_obj_new_int_from_uint(id);
   118	
   119	    return id;
   120	}

   141	void proxy_c_free_obj(uint32_t c_ref) {
   142	    if (c_ref >= PROXY_C_REF_NUM_STATIC) {
   143	        // Remove the object from proxy_c_dict if the c_ref in that dict corresponds to this object.
   144	        // (It may be that this object exists in the dict but with a different c_ref from a more
   145	        // recent proxy of this object.)
   146	        mp_obj_t obj_key = mp_obj_new_int_from_uint((uintptr_t)proxy_c_get_obj(c_ref));
   147	        mp_map_elem_t *elem = mp_map_lookup(mp_obj_dict_get_map(MP_STATE_PORT(proxy_c_dict)), obj_key, MP_MAP_LOOKUP);
   148	        if (elem != NULL && mp_obj_int_get_truncated(elem->value) == c_ref) {
   149	            mp_map_lookup(mp_obj_dict_get_map(MP_STATE_PORT(proxy_c_dict)), obj_key, MP_MAP_LOOKUP_REMOVE_IF_FOUND);
   150	        }

   153	        ((mp_obj_list_t *)MP_OBJ_TO_PTR(MP_STATE_PORT(proxy_c_ref)))->items[c_ref] = MP_OBJ_NULL;
   154	        proxy_c_ref_next = MIN(proxy_c_ref_next, c_ref);

   159	    if (value[0] == PROXY_KIND_JS_UNDEFINED) {

   163	    } else if (value[0] == PROXY_KIND_JS_BOOLEAN) {

   181	void proxy_convert_mp_to_js_obj_cside(mp_obj_t obj, uint32_t *out) {
   182	    uint32_t kind;
   183	    int js_ref;
   184	    if (obj == MP_OBJ_NULL) {
   185	        kind = PROXY_KIND_MP_NULL;
   186	    } else if (obj == mp_const_none) {
   187	        kind = PROXY_KIND_MP_NONE;
   188	    } else if (mp_obj_is_bool(obj)) {
   189	        kind = PROXY_KIND_MP_BOOL;
   190	        out[1] = mp_obj_is_true(obj);
   191	    } else if (mp_obj_is_int(obj)) {
   192	        kind = PROXY_KIND_MP_INT;
   193	        out[1] = mp_obj_get_int_truncated(obj); // TODO support big int
   194	    } else if (mp_obj_is_float(obj)) {
   195	        kind = PROXY_KIND_MP_FLOAT;
   196	        *(double *)&out[1] = mp_obj_get_float(obj);
   197	    } else if (mp_obj_is_str(obj)) {
   198	        kind = PROXY_KIND_MP_STR;
   199	        size_t len;
   200	        const char *str = mp_obj_str_get_data(obj, &len);
   201	        out[1] = len;
   202	        out[2] = (uintptr_t)str;
   203	    } else if (obj == mp_const_undefined) {
   204	        kind = PROXY_KIND_MP_JSPROXY;
   205	        out[1] = 1;
   206	    } else if (mp_obj_is_jsproxy(obj)) {
   207	        kind = PROXY_KIND_MP_JSPROXY;
   208	        out[1] = mp_obj_jsproxy_get_ref(obj);
   209	    } else if ((js_ref = proxy_c_check_existing(obj)) >= 0) {
   210	        kind = PROXY_KIND_MP_EXISTING;
   211	        out[1] = js_ref;
   212	    } else if (mp_obj_get_type(obj) == &mp_type_JsException) {
   213	        mp_obj_exception_t *exc = MP_OBJ_TO_PTR(obj);
   214	        if (exc->args->len > 0 && mp_obj_is_jsproxy(exc->args->items[0])) {
   215	            kind = PROXY_KIND_MP_JSPROXY;
   216	            out[1] = mp_obj_jsproxy_get_ref(exc->args->items[0]);
   217	        } else {
   218	            kind = PROXY_KIND_MP_OBJECT;
   219	            out[1] = proxy_c_add_obj(obj);
   220	        }
   221	    } else {
   222	        if (mp_obj_is_callable(obj)) {
   223	            kind = PROXY_KIND_MP_CALLABLE;
   224	        } else if (mp_obj_is_type(obj, &mp_type_gen_instance)) {
   225	            kind = PROXY_KIND_MP_GENERATOR;
   226	        } else {
   227	            kind = PROXY_KIND_MP_OBJECT;
   228	        }
   229	        out[1] = proxy_c_add_obj(obj);
   230	    }
   231	    out[0] = kind;
   232	}
