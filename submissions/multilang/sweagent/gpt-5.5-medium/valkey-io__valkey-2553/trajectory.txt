  1058	/* Must be called at the start of the timeProc as it measures the delay from the end of the previous
  1059	 * timeProc invocation when performing the computation. */
  1060	static int computeDefragCycleUs(void) {
  1061	    long dutyCycleUs;
  1062	
  1063	    int targetCpuPercent = server.active_defrag_cpu_percent;
  1064	    serverAssert(targetCpuPercent > 0 && targetCpuPercent < 100);
  1065	
  1066	    static int prevCpuPercent = 0; // STATIC - this persists
  1067	    if (targetCpuPercent != prevCpuPercent) {
  1068	        /* If the targetCpuPercent changes, the value might be different from when the last wait
  1069	         *  time was computed.  In this case, don't consider wait time.  (This is really only an
  1070	         *  issue in crazy tests that dramatically increase CPU while defrag is running.) */
  1071	        defrag.timeproc_end_time = 0;
  1072	        prevCpuPercent = targetCpuPercent;
  1073	    }
  1074	
  1075	    // Given when the last duty cycle ended, compute time needed to achieve the desired percentage.
  1076	    if (defrag.timeproc_end_time == 0) {

  1078	        defrag.timeproc_overage_us = 0;
  1079	        dutyCycleUs = server.active_defrag_cycle_us;
  1080	    } else {
  1081	        long waitedUs = getMonotonicUs() - defrag.timeproc_end_time;
  1082	        /* Given the elapsed wait time between calls, compute the necessary duty time needed to
  1083	         *  achieve the desired CPU percentage.
  1084	         *  With:  D = duty time, W = wait time, P = percent
  1085	         *  Solve:    D          P
  1086	         *          -----   =  -----
  1087	         *          D + W       100
  1088	         *  Solving for D:
  1089	         *     D = P * W / (100 - P)
  1090	         *
  1091	         * Note that dutyCycleUs addresses starvation.  If the wait time was long, we will compensate
  1092	         *  with a proportionately long duty-cycle.  This won't significantly affect perceived
  1093	         *  latency, because clients are already being impacted by the long cycle time which caused
  1094	         *  the starvation of the timer. */
  1095	        dutyCycleUs = targetCpuPercent * waitedUs / (100 - targetCpuPercent);
  1096	
  1097	        // Also adjust for any accumulated overage.
  1098	        dutyCycleUs -= defrag.timeproc_overage_us;
  1099	        defrag.timeproc_overage_us = 0;
  1100	
  1101	        if (dutyCycleUs < server.active_defrag_cycle_us) {
  1102	            /* We never reduce our cycle time, that would increase overhead.  Instead, we track this
  1103	             *  as part of the overage, and increase wait time between cycles. */
  1104	            defrag.timeproc_overage_us = server.active_defrag_cycle_us - dutyCycleUs;
  1105	            dutyCycleUs = server.active_defrag_cycle_us;
  1106	        }
  1107	    }
  1108	    return dutyCycleUs;
  1109	}

  1112	/* Must be called at the end of the timeProc as it records the timeproc_end_time for use in the next
  1113	 * computeDefragCycleUs computation. */
  1114	static int computeDelayMs(monotime intendedEndtime) {
  1115	    defrag.timeproc_end_time = getMonotonicUs();
  1116	    long overage = defrag.timeproc_end_time - intendedEndtime;
  1117	    defrag.timeproc_overage_us += overage; // track over/under desired CPU
  1118	    /* Allow negative overage (underage) to count against existing overage, but don't allow
  1119	     * underage (from short stages) to be accumulated.  */
  1120	    if (defrag.timeproc_overage_us < 0) defrag.timeproc_overage_us = 0;
  1121	
  1122	    int targetCpuPercent = server.active_defrag_cpu_percent;
  1123	    serverAssert(targetCpuPercent > 0 && targetCpuPercent < 100);
  1124	
  1125	    // Given the desired duty cycle, what inter-cycle delay do we need to achieve that?
  1126	    // We want to achieve a specific CPU percent.  To do that, we can't use a skewed computation.
  1127	    // Example, if we run for 1ms and delay 10ms, that's NOT 10%, because the total cycle time is 11ms.
  1128	    // Instead, if we rum for 1ms, our total time should be 10ms.  So the delay is only 9ms.
  1129	    long totalCycleTimeUs = server.active_defrag_cycle_us * 100 / targetCpuPercent;

  1131	    // Only increase delay by the fraction of the overage that would be non-duty-cycle
  1132	    delayUs += defrag.timeproc_overage_us * (100 - targetCpuPercent) / 100;
  1133	    if (delayUs < 0) delayUs = 0;
  1134	    long delayMs = delayUs / 1000; // round down
  1135	    return delayMs;
  1136	}
