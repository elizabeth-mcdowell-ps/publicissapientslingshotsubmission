template <typename VectorType>
chaiscript::ModulePtr register_vector_of_unique_ptr(const std::string& type)
{

    using namespace chaiscript;
    ModulePtr m = std::make_shared<Module>();
    m->add(user_type<VectorType>(), type);

    typedef VectorType::reference(VectorType::*frontptr)();
    typedef VectorType::const_reference(VectorType::*constfrontptr)() const;

    m->add(fun(static_cast<frontptr>(&VectorType::front)), "front");
    m->add(fun(static_cast<constfrontptr>(&VectorType::front)), "front");


    //bootstrap::standard_library::back_insertion_sequence_type<VectorType>(type, m); //breaks unique_ptrs
    //bootstrap::standard_library::sequence_type<VectorType>(type, m); //breaks unique_ptrs
    bootstrap::standard_library::random_access_container_type<VectorType>(type, m);
    bootstrap::standard_library::container_type<VectorType>(type, m);
    bootstrap::standard_library::default_constructible_type<VectorType>(type, m);
    //bootstrap::standard_library::assignable_type<VectorType>(type, m); //breaks unique_ptrs
    bootstrap::standard_library::input_range_type<VectorType>(type, m);

    if (typeid(VectorType) == typeid(std::vector<Boxed_Value>))
    {
        m->eval(R"(
                    def Vector::`==`(Vector rhs) {
                       if ( rhs.size() != this.size() ) {

                       } else {

                         while (!r1.empty())

                           if (!eq(r1.front(), r2.front()))
                           {
                             return false;
                           }
                           r1.pop_front();
                           r2.pop_front();
                         }
                         true;
                       }
                   } )"
                   );
    }

  return m;
