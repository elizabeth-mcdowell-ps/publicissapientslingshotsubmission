   167	// stripAA55Header validates the AA55 response frame and returns the bare
   168	// payload (without the 5-byte header and trailing 2-byte CRC).
   169	// buf[2] is the inverter source address and varies by family — only the
   170	// AA 55 magic bytes and function code 0x03 are validated.
   171	func stripAA55Header(buf []byte) ([]byte, error) {
   172	        if len(buf) < 6 || buf[0] != 0xAA || buf[1] != 0x55 || buf[3] != 0x03 {
   173	                return nil, errors.New("invalid response header")
   174	        }
   175	        byteCount := int(buf[4])
   176	        if len(buf) < 5+byteCount+2 {
   177	                return nil, errors.New("short response")
   178	        }
   179	        return buf[5 : 5+byteCount], nil
   180	}

   182	// decodeAt extracts an integer at the given byte offset of payload and
   183	// interprets it according to decode.
   184	func decodeAt(payload []byte, offset int, decode string) (float64, error) {
   185	        switch decode {
   186	        case "int32be":
   187	                if len(payload) < offset+4 {
   188	                        return 0, fmt.Errorf("payload too short for int32be at offset %d (len=%d)", offset, len(payload))
   189	                }
   190	                return float64(int32(binary.BigEndian.Uint32(payload[offset:]))), nil
   191	        case "uint32be":
   192	                if len(payload) < offset+4 {
   193	                        return 0, fmt.Errorf("payload too short for uint32be at offset %d (len=%d)", offset, len(payload))
   194	                }
   195	                return float64(binary.BigEndian.Uint32(payload[offset:])), nil
   196	        case "int16be":

   199	                }

   204	                }

   210	// modbusCRC16 computes the Modbus CRC-16 (little-endian byte order).
   211	func modbusCRC16(data []byte) []byte {
   212	        crc := uint16(0xFFFF)
   213	        for _, b := range data {
   214	                crc ^= uint16(b)
   215	                for i := 0; i < 8; i++ {
   216	                        if crc&0x0001 != 0 {
   217	                                crc = (crc >> 1) ^ 0xA001
   218	                        } else {
   219	                                crc >>= 1
   220	                        }
   221	                }
   222	        }
   223	        return []byte{byte(crc & 0xFF), byte(crc >> 8)}
   224	}

// evcc source plugin.
//
// The inverter speaks a simple request/response protocol over UDP port 8899:
//
//	Request:  [6-byte Modbus PDU body] [Modbus CRC-16, little-endian]
//	Response: AA 55 [src] 03 [byteCount] [payload…] [CRC]
//
// src varies by inverter family (0x7F for DT/DNS, 0xF7 for ET/EH/BT/BH);
// only the AA 55 magic bytes and function code 0x03 are validated.
//

	conn   *net.UDPConn
