    63	// createCSIKeyring returns four values: the cephClientID with the keyCount as a suffix,
    64	// the cephClient key value, a boolean indicating whether the key should be rotated, and an error.
    65	// It calls the method `getCsiKeyRotationInfo` to find any Ceph clients matching the CSI keyBase
    66	// and the maximum keyGen. If no matching Ceph clients are found, it returns keyGen as 0 and an
    67	// empty list. If keys are found, a new key is generated with an incremented keyGen as a suffix,
    68	// and the list of matching Ceph clients will be deleted later in the method.
    69	// It fetches the current `interpretedCephxStatus` and keyGen from the CephCluster,
    70	// and then calls the method `ShouldRotateCephxKeys` to determine if the old key should be rotated.
    71	// Based on this, it updates the cephxStatus, generates a new client name, and creates a new key.
    72	func createCSIKeyring(
    73	        context *clusterd.Context, clusterInfo *client.ClusterInfo, cephCluster *cephv1.CephCluster,
    74	        s *keyring.SecretStore, csiKeyrigName string, keyCaps []string,
    75	) (string, string, int, bool, error) {
    76	        logger.Debugf("starting CSI key generation for cluster in namespace %q", clusterInfo.Namespace)
    77	
    78	        currentMaxKeyGen, allKeyWithBaseName, err := getCsiKeyRotationInfo(context, clusterInfo, csiKeyrigName)
    79	        if err != nil {
    80	                return "", "", 0, false, errors.Wrapf(err, "failed to determine key rotation info for CSI key %s", csiKeyrigName)
    81	        }
    82	
    83	        // corner cases could have the status KeyGeneration out of sync with what was determined above
    84	        // apply the current max key gen to the current status to find the definitive current status
    85	        interpretedCephxStatus := cephCluster.Status.Cephx.CSI.DeepCopy()
    86	        interpretedCephxStatus.KeyGeneration = uint32(currentMaxKeyGen) //nolint:gosec // disable G115
    87	
    88	        // determine shouldRotate based on 'definitive' status
    89	        shouldRotate, err := keyring.ShouldRotateCephxKeys(
    90	                cephCluster.Spec.Security.CephX.CSI.CephxConfig,
    91	                clusterInfo.CephVersion, clusterInfo.CephVersion,
    92	                interpretedCephxStatus.CephxStatus,
    93	        )
    94	        if err != nil {
    95	                return "", "", 0, shouldRotate, errors.Wrap(err, "failed to call `shouldRotateCephxKeys` during CSI key rotation")
    96	        }
    97	
    98	        // generate the future cephx status to determine what the new key gen should be
    99	        desiredCephxStatus := keyring.UpdatedCephxStatus(shouldRotate,
   100	                cephCluster.Spec.Security.CephX.CSI.CephxConfig,
   101	                clusterInfo.CephVersion,
   102	                interpretedCephxStatus.CephxStatus)

   106

   108	        key, err := s.GenerateKey(latestClientId, keyCaps)
   109	        if err != nil {
   110	                return "", "", 0, shouldRotate, errors.Wrapf(err, "failed to check if keys should to be rotated for CSI key %s", csiKeyrigName)
   111	        }
   112	
   113	        keysDeleted, err := deleteOldKeyGen(context, clusterInfo, allKeyWithBaseName, cephCluster.Spec.Security.CephX.CSI.KeepPriorKeyCountMax)

   116	        }

   118	        // add the new key to the list of all keys
