   174	/// Find the start and end indices of the last contiguous block of Xs.
   175	///
   176	/// If no contiguous block of at least three Xs could be found, this
   177	/// function returns `None`.
   178	///
   179	/// # Examples

   186	    let j = s.rfind("XXX")? + 3;
   187	    let i = s[..j].rfind(|c| c != 'X').map_or(0, |i| i + 1);

   405	fn dry_exec(tmpdir: &Path, prefix: &str, rand: usize, suffix: &str) -> UResult<PathBuf> {
   406	    let len = prefix.len() + suffix.len() + rand;
   407	    let mut buf = Vec::with_capacity(len);
   408	    buf.extend(prefix.as_bytes());
   409	    buf.extend(iter::repeat_n(b'X', rand));
   410	    buf.extend(suffix.as_bytes());
   411	
   412	    // Randomize.
   413	    let bytes = &mut buf[prefix.len()..prefix.len() + rand];
   414	    rand::rng().fill(bytes);
   415	    for byte in bytes {
   416	        *byte = match *byte % 62 {
   417	            v @ 0..=9 => v + b'0',
   418	            v @ 10..=35 => v - 10 + b'a',
   419	            v @ 36..=61 => v - 36 + b'A',
   420	            _ => unreachable!(),
   421	        }
   422	    }
   423	    // We guarantee utf8.
   424	    let buf = String::from_utf8(buf).unwrap();
   425	    let tmpdir = Path::new(tmpdir).join(buf);
   426	    Ok(tmpdir)
   427	}
